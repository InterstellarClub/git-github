git clone [--template=<template_directory>]

	  [-o <name>] [-b <name>] [-u <upload-pack>] [--reference <repository>]

	  [--depth <depth>] [--[no-]single-branch] [--no-tags]
	  [--recurse-submodules[=<pathspec>]] [--[no-]shallow-submodules]
	  [--jobs <n>] [--] <repository> [<directory>]
